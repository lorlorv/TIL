	static boolean check(String s) {
		Stack<Character> stack = new Stack<>();
		for(int i = 0; i < s.length(); i++) {
			char c= s.charAt(i);
			
			switch(c) {
			case '(': case'[':
				stack.push(c);
				break;
			case ')': case ']':
				if(stack.empty()) {
					return false;
				}
				else {
					char check = stack.pop();
					if(check == '(' && c == ']' || check == '[' && c == ')') {
						return false;
					}
				}
				break;
			}			
		}
		
		if(!stack.empty())
			return false;
		return true;
	}
